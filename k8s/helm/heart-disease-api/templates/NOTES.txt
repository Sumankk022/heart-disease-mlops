1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
  https://{{ index .Values.ingress.hosts 0 }}.{{ index .Values.ingress.hosts 0 }}.local/
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "heart-disease-api.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "heart-disease-api.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "heart-disease-api.fullname" . }} --template "{.status.loadBalancer.ingress[0].ip}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "heart-disease-api.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8000 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8000:$CONTAINER_PORT
{{- end }}

2. Test the API:
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "heart-disease-api.fullname" . }} 8000:{{ .Values.service.port }} &
  curl http://localhost:8000/health
  curl -X POST http://localhost:8000/predict \
    -H 'Content-Type: application/json' \
    -d '{"age":63,"sex":1,"cp":1,"trestbps":145,"chol":233,"fbs":1,"restecg":2,"thalach":150,"exang":0,"oldpeak":2.3,"slope":3,"ca":0,"thal":6}'

3. View logs:
  kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "heart-disease-api.name" . }},app.kubernetes.io/instance={{ .Release.Name }} -f

4. View API documentation:
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "heart-disease-api.fullname" . }} 8000:{{ .Values.service.port }} &
  Open http://localhost:8000/docs in your browser

5. Check deployment status:
  kubectl describe deployment -n {{ .Release.Namespace }} {{ include "heart-disease-api.fullname" . }}
  kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "heart-disease-api.name" . }}
